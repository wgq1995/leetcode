# 题目描述
    指定背包容量b，物品重量数组w，物品价值数字v，求不超过容量的情况下，最大的价值，每个物品只能选择一次
# 代码：

```
def solution(v, w, b):
    l1 = [0] * (b + 1)
    l2 = []
    for p in range(len(v) - 1, -1, -1):
        for cur_w in range(b+1):
            res1 = l1[cur_w]
            if cur_w + w[p] > b:
                l2.append(res1)
            else:
                l2.append(max(res1, v[p] + l1[cur_w + w[p]]))
        l1 = l2
        l2 = []
    return l1[0]


if __name__ == '__main__':
    v = [2, 3, 5, 7]
    w = [3, 4, 5, 6]
    b = 12
    print(solution(v, w, b))
```
